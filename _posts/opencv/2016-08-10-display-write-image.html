Crear archivo demo1.py {% highlight python linenos %} import cv2 #Cargar imagen img = cv2.imread('img/adilen.jpg', 0) #Mostrar cv2.imshow('image', img) #Espera KeyPress k = cv2.waitKey(0) & 0xFF if k == 27: #ESC para cerrar cv2.destroyAllWindows() elif k == ord('s'): #s para guardar cv2.imwrite('img/miss.jpg', img) cv2.destroyAllWindows() {% endhighlight %}
